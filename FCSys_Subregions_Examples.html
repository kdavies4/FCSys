Test a subregion

Information

Extends from Modelica.Icons.Example (Icon for runnable examples).

Parameters

SubregionClosedbC2redeclare FCSys.BCs.FaceBus....

Modelica definition

model SubregionH2 "Test a subregion"
  extends Modelica.Icons.Example;
  parameter Boolean inclC=false "Carbon (C)";
  parameter Boolean inclC19HF37O5S=false 
    "Nafion sulfonate (C19HF37O5S)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water (H2O)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  FCSys.Subregions.Subregion subregion(
    L={1,1,1}*U.cm,
    graphite(
      final inclC=inclC,
      final 'incle-'='incle-',
      C(V_IC=subregion.V/4)),
    ionomer(
      final inclC19HF37O5S=inclC19HF37O5S,
      final 'inclH+'='inclH+',
      C19HF37O5S(V_IC=subregion.V/4)),
    inclLinY=false,
    inclLinZ=false,
    inclYFaces=false,
    inclZFaces=false,
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2));
  inner FCSys.BCs.Environment environment(analysis=true);
  replaceable FCSys.BCs.FaceBus.SubregionClosed bC1(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite(final inclC=inclC, final 'incle-'='incle-'),
    ionomer(final inclC19HF37O5S=inclC19HF37O5S, final 'inclH+'='inclH+')) 
    constrainedby FCSys.BCs.FaceBus.Subregion;

  replaceable FCSys.BCs.FaceBus.SubregionClosed bC2(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite(final inclC=inclC, final 'incle-'='incle-'),
    ionomer(final inclC19HF37O5S=inclC19HF37O5S, final 'inclH+'='inclH+')) 
    constrainedby FCSys.BCs.FaceBus.Subregion;

equation 
  connect(bC2.face, subregion.xPositive);

  connect(bC1.face, subregion.xNegative);

end SubregionH2;

model SubregionHOR 
  "Test a subregion with the hydrogen oxidation reaction and the essential species for it (C, C19HF37O5S, e-, H2, and H+)"

  extends SubregionH2(
    inclC=true,
    inclC19HF37O5S=true,
    'incle-'=true,
    inclH2=true,
    'inclH+'=true,
    bC2(
      gas(H2(redeclare FCSys.BCs.Face.Material.Density material(spec(k=U.atm/(298.15
                  *U.K))))),
      graphite(C(redeclare FCSys.BCs.Face.Thermal.Temperature thermal(spec(k=
                  environment.T))), 'e-'(redeclare FCSys.BCs.Face.Material.Current
            material(redeclare Modelica.Blocks.Sources.Ramp spec(duration=1000,
                height=2*U.A)))),
      ionomer('H+'(redeclare FCSys.BCs.Face.Material.Density material(spec(k(
                  start=U.atm/(298.15*U.K))))))));

  extends Modelica.Icons.UnderConstruction;

end SubregionHOR;

model SubregionsH2 "Test a one-dimensional array of subregions"
  extends Modelica.Icons.Example;
  extends Modelica.Icons.UnderConstruction;
  // **fails sim
  parameter Integer n_x=0 
    "Number of discrete subregions along the x axis, besides the 2 side subregions";
  parameter Boolean inclC=false "Carbon (C)";
  parameter Boolean inclC19HF37O5S=false 
    "Nafion sulfonate (C19HF37O5S)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water (H2O)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  FCSys.Subregions.Subregion subregion1(
    L={1,1,1}*U.cm,
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2O(
        p_IC=environment.p + 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      N2(
        p_IC=environment.p + 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      O2(
        p_IC=environment.p + 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      H2(
        p_IC=environment.p + 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    graphite(
      final inclC=inclC,
      final 'incle-'='incle-',
      C(V_IC=subregion1.V/4),
      'e-'(
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    ionomer(
      final inclC19HF37O5S=inclC19HF37O5S,
      final 'inclH+'='inclH+',
      C19HF37O5S(V_IC=subregion1.V/4),
      'H+'(
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    inclLinY=false,
    inclLinZ=false,
    inclYFaces=false,
    inclZFaces=false);

  FCSys.Subregions.Subregion subregions[n_x](
    each L={1,1,1}*U.cm,
    gas(
      each final inclH2=inclH2,
      each final inclH2O=inclH2O,
      each final inclN2=inclN2,
      each final inclO2=inclO2,
      H2(
        each p_IC=environment.p,
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true)),
      H2O(
        each p_IC=environment.p,
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true)),
      N2(
        each p_IC=environment.p,
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true)),
      O2(
        each p_IC=environment.p,
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true))),
    graphite(
      each final inclC=inclC,
      each final 'incle-'='incle-',
      C(each V_IC=subregions[1].V/4),
      'e-'(
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true))),
    ionomer(
      each final inclC19HF37O5S=inclC19HF37O5S,
      each final 'inclH+'='inclH+',
      C19HF37O5S(each V_IC=subregions[1].V/4),
      'H+'(
        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true))),
    each inclLinY=false,
    each inclLinZ=false,
    each inclYFaces=false,
    each inclZFaces=false) if n_x > 0;
  FCSys.Subregions.Subregion subregion2(
    L={1,1,1}*U.cm,
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(
        p_IC=environment.p - 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      H2O(
        p_IC=environment.p - 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      N2(
        p_IC=environment.p - 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true)),
      O2(
        p_IC=environment.p - 1*U.kPa,
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    graphite(
      final inclC=inclC,
      final 'incle-'='incle-',
      C(V_IC=subregion2.V/4),
      'e-'(
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    ionomer(
      final inclC19HF37O5S=inclC19HF37O5S,
      final 'inclH+'='inclH+',
      C19HF37O5S(V_IC=subregion2.V/4),
      'H+'(
        yNegative(inviscidX=true),
        yPositive(inviscidX=true),
        zNegative(inviscidX=true),
        zPositive(inviscidX=true))),
    inclLinY=false,
    inclLinZ=false,
    inclYFaces=false,
    inclZFaces=false);

  inner FCSys.BCs.Environment environment(analysis=true);
  replaceable FCSys.BCs.FaceBus.SubregionClosed bC1 constrainedby 
    FCSys.BCs.FaceBus.Subregion(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite(
      final inclC=inclC,
      final 'incle-'='incle-',
      C(thermoOpt=ThermoOpt.ClosedDiabatic)),
    ionomer(
      final inclC19HF37O5S=inclC19HF37O5S,
      final 'inclH+'='inclH+',
      C19HF37O5S(thermoOpt=ThermoOpt.ClosedDiabatic)));
  replaceable FCSys.BCs.FaceBus.SubregionClosed bC2 constrainedby 
    FCSys.BCs.FaceBus.Subregion(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite(
      final inclC=inclC,
      final 'incle-'='incle-',
      C(thermoOpt=ThermoOpt.ClosedDiabatic)),
    ionomer(
      final inclC19HF37O5S=inclC19HF37O5S,
      final 'inclH+'='inclH+',
      C19HF37O5S(thermoOpt=ThermoOpt.ClosedDiabatic)));

equation 
  connect(bC1.face, subregion1.xNegative);
  connect(subregion1.xPositive, subregions[1].xNegative);

  for i in 1:n_x - 1 loop
    connect(subregions[i].xPositive, subregions[i + 1].xNegative) 
      "Not shown on the diagram";
  end for;
  if n_x == 0 then
    connect(subregion1.xPositive, subregion2.xNegative) 
      "Not shown on the diagram";
  end if;
  connect(subregions[n_x].xPositive, subregion2.xNegative);
  connect(bC2.face, subregion2.xPositive);

end SubregionsH2;

        each yNegative(inviscidX=true),
        each yPositive(inviscidX=true),
        each zNegative(inviscidX=true),
        each zPositive(inviscidX=true))),
    each inclLinY=false,
    each inclLinZ=false,
    each inclYFaces=false,
    each inclZFaces=false) if n_x > 0;

  FCSys.Subregions.Subregion subregion2(

      inclH2=false,
      inclH2O=true,
      inclN2=false,

      inclH2=true,
      inclH2O=false,
      inclN2=false,
      inclO2=false), graphite(
      inclC=true,
      'incle-'=true,
      C(redeclare FCSys.BCs.Face.Thermal.Temperature thermal(spec(k=environment.T))),

      'e-'(redeclare FCSys.BCs.Face.Material.Current material(redeclare Modelica.Blocks.Sources.Ramp
            spec(duration=1000, height=-2*U.A)))));
  replaceable FCSys.BCs.FaceBus.SubregionFlow bC2 constrainedby 
    FCSys.BCs.FaceBus.Subregion(gas(
      inclH2=false,
      inclH2O=true,
      inclN2=false,
      inclO2=true), graphite(
      inclC=true,
      'incle-'=true,
      C(redeclare FCSys.BCs.Face.Thermal.Temperature thermal(spec(k=environment.T))),

            spec(duration=1000, height=2*U.A)))));

  connect(subregion1.xPositive, subregions[1].xNegative);

  for i in 1:n_x - 1 loop
    connect(subregions[i].xPositive, subregions[i + 1].xNegative) 
      "Not shown on the diagram";
  end for;
  if n_x == 0 then
    connect(subregion1.xPositive, subregion2.xNegative) 
      "Not shown on the diagram";
  end if;
  connect(subregions[n_x].xPositive, subregion2.xNegative);
  connect(bC2.face, subregion2.xPositive);

  connect(ground.face, subregion1.xNegative);

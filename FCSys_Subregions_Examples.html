model Subregion 
  "Evaluate a single subregion, with H2 by default"

  extends Modelica.Icons.Example;

  parameter Boolean 'inclC+'=false "Carbon plus (C+)";
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate (C19HF37O5S-)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water vapor (H2O)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  FCSys.Subregions.Subregion subregion(
    L={1,1,1}*U.cm,
    inclTransY=false,
    inclTransZ=false,
    inclFacesY=false,
    inclFacesZ=false,
    inclFacesX=true,
    graphite(
      final 'inclC+'='inclC+',
      final 'incle-'='incle-',
      'C+'(V_IC=subregion.V/4)),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    liquid(H2O(V_IC=subregion.V/4)),
    ionomer(
      final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      final 'inclH+'='inclH+',
      'C19HF37O5S-'(V_IC=subregion.V/4)));

  inner FCSys.Conditions.Environment environment(analysis=true);

end Subregion;

model SubregionHOR 
  "Test a subregion with the hydrogen oxidation reaction and the essential species for it (C+, C19HF37O5S, e-, H2, and H+)"

  extends Examples.Subregion(
    'inclC+'=true,
    'inclC19HF37O5S-'=true,
    'incle-'=true,
    'inclH+'=true,
    inclH2=true,
    subregion(
      gas(H2(
          initMaterial=FCSys.Subregions.Species.BaseClasses.InitScalar.None,
          chemical(Ndot(start=0, fixed=true)),
          consEnergy=FCSys.Subregions.Species.BaseClasses.Conservation.IC)),
      graphite(
        reduceTemp=false,
        'C+'(consEnergy=FCSys.Subregions.Species.BaseClasses.Conservation.IC),
        'e-'(consEnergy=FCSys.Subregions.Species.BaseClasses.Conservation.IC)),

      ionomer(
        reduceTemp=false,
        'C19HF37O5S-'(consEnergy=FCSys.Subregions.Species.BaseClasses.Conservation.IC),

        'H+'(consEnergy=FCSys.Subregions.Species.BaseClasses.Conservation.IC))));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated negativeBC(
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal(redeclare Modelica.Blocks.Sources.Ramp source(
            duration=100,
            height=3*U.N,
            startTime=10)))),
    ionomer(
      'inclC19HF37O5S-'=false,
      final 'inclH+'='inclH+',
      'H+'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Density
          material(source(y=environment.p/environment.T)))));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated positiveBC(
    graphite(
      'inclC+'=false,
      final 'incle-'='incle-',
      'e-'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)),
    ionomer(
      'inclC19HF37O5S-'=false,
      final 'inclH+'='inclH+',
      'H+'(redeclare Conditions.ByConnector.Face.Single.Translational.Force
          normal)),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2));

  extends Modelica.Icons.UnderConstruction;

equation 
  connect(negativeBC.face, subregion.xNegative);

  connect(subregion.xPositive, positiveBC.face);

end SubregionHOR;

model SubregionPipeFlow
  import assert = FCSys.BaseClasses.Utilities.assertEval;
  extends Examples.Subregion(subregion(
      L={100,1,1}*U.mm,
      inclFacesY=true,
      inclFacesZ=true));

  parameter Q.Pressure Deltap=-10*U.Pa 
    "Prescribed gas pressure difference (Δp)";

  output Q.Length D=2*subregion.A[Axis.x]/(subregion.L[Axis.y] + subregion.L[
      Axis.z]);
  output Q.Number Re=subregion.gas.H2.phi[Axis.x]*D*subregion.gas.H2.zeta*
      subregion.gas.H2.Data.m*subregion.gas.H2.rho "Reynolds number";
  output Q.Pressure Deltap_Poiseuille=-32*subregion.L[Axis.x]*subregion.gas.H2.phi[
      Axis.x]/(D^2*subregion.gas.H2.zeta) 
    "Pressure difference according to Poiseuille's law";

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC1(
    graphite('inclC+'=false, final 'incle-'='incle-'),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'),
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2(redeclare FCSys.Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      H2O(redeclare FCSys.Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      N2(redeclare FCSys.Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Pressure
          material(source(y=environment.p - Deltap/2))),
      O2(redeclare FCSys.Conditions.ByConnector.Face.Single.Translational.Force
          normal, redeclare FCSys.Conditions.ByConnector.Face.Single.Material.Pressure

model Subregions 
  "Test a one-dimensional array of subregions with an initial pressure difference (C+ and H2 included by default)"
  extends Modelica.Icons.Example;

  parameter Integer n_x=0 
    "Number of discrete subregions along the x axis, besides the 2 side subregions (nx)";
  parameter Q.Pressure Deltap_IC=100*U.Pa 
    "Initial pressure difference (ΔpIC)";
  parameter Boolean 'inclC+'=true "Carbon plus (C+)";
  parameter Boolean 'inclC19HF37O5S-'=false 
    "Nafion sulfonate (C19HF37O5S)";
  parameter Boolean 'incle-'=false "Electrons (e-)";
  parameter Boolean inclH2=true "Hydrogen (H2)";
  parameter Boolean inclH2O=false "Water vapor (H2O)";
  parameter Boolean 'inclH+'=false "Protons (H+)";
  parameter Boolean inclN2=false "Nitrogen (N2)";
  parameter Boolean inclO2=false "Oxygen (O2)";

  FCSys.Subregions.Subregion subregion1(
    L={1,1,1}*U.cm,
    inclFacesY=false,
    inclFacesZ=false,
    inclTransY=false,
    inclTransZ=false,
    gas(
      reduceVel=true,
      reduceTemp=true,
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2O(p_IC=environment.p + Deltap_IC/2),
      N2(p_IC=environment.p + Deltap_IC/2),
      O2(p_IC=environment.p + Deltap_IC/2),
      H2(p_IC=environment.p + Deltap_IC/2)),
    graphite(
      reduceTemp=true,
      final 'inclC+'='inclC+',
      final 'incle-'='incle-',
      'C+'(V_IC=subregion1.V/1000)),
    ionomer(
      reduceTemp=true,
      final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      final 'inclH+'='inclH+',
      'C19HF37O5S-'(V_IC=subregion1.V/1000)),
    liquid(H2O(V_IC=subregion1.V/4)));

  FCSys.Subregions.Subregion subregions[n_x](
    each L={1,1,1}*U.cm,
    each inclTransY=false,
    each inclTransZ=false,
    each inclFacesY=false,
    each inclFacesZ=false,
    graphite(
      each reduceTemp=true,
      each final 'inclC+'='inclC+',
      each final 'incle-'='incle-',
      'C+'(each V_IC=subregions[1].V/1000)),
    ionomer(
      each reduceTemp=true,
      each final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      each final 'inclH+'='inclH+',
      'C19HF37O5S-'(each V_IC=subregions[1].V/1000)),
    gas(
      each reduceTemp=true,
      each reduceVel=true,
      each final inclH2=inclH2,
      each final inclH2O=inclH2O,
      each final inclN2=inclN2,
      each final inclO2=inclO2,
      H2(p_IC={environment.p + Deltap_IC/2 - i*Deltap_IC/(n_x + 1) for i in 1:
            n_x}),
      H2O(p_IC=subregions.gas.H2.p_IC),
      N2(p_IC=subregions.gas.H2.p_IC),
      O2(p_IC=subregions.gas.H2.p_IC)),
    liquid(H2O(each V_IC=subregions[1].V/4))) if n_x > 0;

  FCSys.Subregions.Subregion subregion2(
    L={1,1,1}*U.cm,
    inclFacesY=false,
    inclFacesZ=false,
    inclTransY=false,
    inclTransZ=false,
    gas(
      reduceVel=true,
      reduceTemp=true,
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2,
      H2O(p_IC=environment.p - Deltap_IC/2),
      N2(p_IC=environment.p - Deltap_IC/2),
      O2(p_IC=environment.p - Deltap_IC/2),
      H2(p_IC=environment.p - Deltap_IC/2)),
    graphite(
      reduceTemp=true,
      final 'inclC+'='inclC+',
      final 'incle-'='incle-',
      'C+'(V_IC=subregion2.V/1000)),
    ionomer(
      reduceTemp=true,
      final 'inclC19HF37O5S-'='inclC19HF37O5S-',
      final 'inclH+'='inclH+',
      'C19HF37O5S-'(V_IC=subregion2.V/1000)));

  inner FCSys.Conditions.Environment environment(analysis=true);

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC1(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite('inclC+'=false, final 'incle-'='incle-'),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'));

  Conditions.ByConnector.FaceBus.Single.FaceBusIsolated BC2(
    gas(
      final inclH2=inclH2,
      final inclH2O=inclH2O,
      final inclN2=inclN2,
      final inclO2=inclO2),
    graphite('inclC+'=false, final 'incle-'='incle-'),
    ionomer('inclC19HF37O5S-'=false, final 'inclH+'='inclH+'));

equation 
  connect(subregion1.xPositive, subregions[1].xNegative);

  for i in 1:n_x - 1 loop
    connect(subregions[i].xPositive, subregions[i + 1].xNegative) 
      "Not shown in the diagram";
  end for;
  if n_x == 0 then
    connect(subregion1.xPositive, subregion2.xNegative) 
      "Not shown in the diagram";
  end if;
  connect(subregions[n_x].xPositive, subregion2.xNegative);

model TestReaction "Test an electrochemical reaction"
  extends Modelica.Icons.Example;
  import FCSys.BaseClasses.Utilities.cartWrap;
  import FCSys.BaseClasses.Utilities.countTrue;
  import FCSys.BaseClasses.Utilities.enumerate;
  import FCSys.BaseClasses.Utilities.index;

  // Assumptions
  // -----------
  // Included components of translational momentum
  parameter Boolean inclTransX=true "X";
  parameter Boolean inclTransY=false "Y";
  parameter Boolean inclTransZ=false "Z";

  Reaction reaction;

  inner Conditions.Environment environment(analysis=true);

  ChemicalExchange chemicalExchange[3](n={-2,-2,1}, m={1,1,1}*U.g/U.mol);
  FCSys.Subregions.Depletion layer1(transSubstrate=true, thermalSubstrate=true);
  FCSys.Subregions.Depletion layer2;
  replaceable Conditions.ByConnector.ChemicalSpecies.Current current(
    redeclare Modelica.Blocks.Sources.Ramp source(height=100*U.A, duration=100),

    final inclTransX=inclTransX,
    final inclTransY=inclTransY,
    final inclTransZ=inclTransZ,
    sT=3000*U.K);

  replaceable Conditions.ByConnector.ChemicalSpecies.Potential potential[2](
    each sT=0,
    each final inclTransX=inclTransX,
    each final inclTransY=inclTransY,
    each final inclTransZ=inclTransZ);
  Conditions.ByConnector.Inert.Inert catalyst(
    thermal(source(y=360*U.K)),
    final inclTransX=inclTransX,
    final inclTransY=inclTransY,
    final inclTransZ=inclTransZ);

  Conditions.ByConnector.Face.Single.Face face1(redeclare Conditions.ByConnector.Face.Single.Translational.Force
      normal);
  Conditions.ByConnector.Face.Single.Face face2(redeclare Conditions.ByConnector.Face.Single.Translational.Force
      normal);

protected 
  final inner parameter Boolean inclTrans[Axis]={inclTransX,inclTransY,
      inclTransZ} 
    "true, if each component of translational momentum is included";
  final inner parameter Integer n_trans=countTrue(inclTrans) 
    "Number of components of translational momentum";
  final inner parameter Integer cartTrans[:]=index(inclTrans) 
    "Cartesian-axis indices of the components of translational momentum";

equation 
  connect(chemicalExchange[1].reaction, reaction.chemical);
  connect(chemicalExchange[2].reaction, reaction.chemical);
  connect(chemicalExchange[3].reaction, reaction.chemical);
  connect(layer2.chemical, reaction.chemical);
  connect(layer1.chemical, reaction.chemical);
  connect(current.chemical, chemicalExchange[3].species);

  connect(potential.chemical, chemicalExchange[1:2].species);
  connect(catalyst.inert, layer1.inert);
  connect(catalyst.inert, layer2.inert);
  connect(face1.face, layer1.face);
  connect(layer2.face, face2.face);
end TestReaction;

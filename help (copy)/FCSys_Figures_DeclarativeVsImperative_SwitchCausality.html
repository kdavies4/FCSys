model Declarative_vi 
  "Declarative-based circuit with voltage in, current out"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;
  Connectors.RealInput v;
  Connectors.RealOutput i;
  Modelica.Electrical.Analog.Basic.Resistor res1(R=R1);
  Modelica.Electrical.Analog.Basic.Capacitor cap(C=C);
  Modelica.Electrical.Analog.Basic.Inductor ind(L=L);
  Modelica.Electrical.Analog.Basic.Ground ground;
  Modelica.Electrical.Analog.Basic.Resistor res2(R=R2);
  Modelica.Electrical.Analog.Sources.SignalVoltage voltageSource;
  Modelica.Electrical.Analog.Sensors.CurrentSensor currentSensor;

equation 
  connect(res2.p, res1.p);
  connect(cap.p, res1.n);

  connect(ind.p, res2.n);
  connect(ind.n, ground.p);
  connect(voltageSource.n, ground.p);
  connect(v, voltageSource.v);
  connect(i, currentSensor.i);
  connect(cap.n, ground.p);
  connect(i, i);
  connect(currentSensor.p, res1.p);
  connect(currentSensor.n, voltageSource.p);
end Declarative_vi;

model Imperative_vi "Imperative circuit with voltage in, current out"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;
  Connectors.RealInput v;
  Connectors.RealOutput i;
  Modelica.Blocks.Continuous.Integrator ind(k=1/L);
  Modelica.Blocks.Math.Gain res1(k=1/R1);
  Modelica.Blocks.Math.Add diff1(k1=-1, k2=1);
  Modelica.Blocks.Math.Add sum(k1=-1, k2=-1);
  Modelica.Blocks.Math.Add diff2(k2=-1);
  Modelica.Blocks.Continuous.Integrator cap(k=1/C);
  Modelica.Blocks.Math.Gain res2(k=R2);

equation 
  connect(diff1.y, res1.u);
  connect(sum.y, i);
  connect(diff2.y, ind.u);
  connect(cap.u, res1.y);
  connect(res2.u, ind.y);

  connect(diff1.u2, v);
  connect(diff2.u1, v);
  connect(ind.y, sum.u2);
  connect(sum.u1, res1.y);
  connect(cap.y, diff1.u1);
  connect(res2.y, diff2.u2);
end Imperative_vi;

model Imperative_iv "Imperative circuit with current in, voltage out"
  extends BaseClasses.Parameters;
  extends FCSys.BaseClasses.Icons.Blocks.Continuous;

public 
  Connectors.RealInput i;
  Connectors.RealOutput v;
  Modelica.Blocks.Math.Gain res1(k=R1);
  Modelica.Blocks.Math.Add sum;
  Modelica.Blocks.Math.Add diff1(k2=-1, k1=-1);
  Modelica.Blocks.Continuous.Integrator cap(k=1/C);
  Modelica.Blocks.Continuous.Integrator ind(k=1/L);
  Modelica.Blocks.Math.Add diff2(k2=-1);
  Modelica.Blocks.Math.Gain res2(k=R2);

equation 
  connect(res2.u, ind.y);
  connect(diff2.y, ind.u);
  connect(i, diff1.u1);
  connect(diff1.u2, ind.y);
  connect(cap.u, diff1.y);
  connect(diff1.y, res1.u);
  connect(cap.y, sum.u1);
  connect(sum.u2, res1.y);
  connect(diff2.u2, res2.y);
  connect(sum.y, diff2.u1);
  connect(sum.y, v);
end Imperative_iv;

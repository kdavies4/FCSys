model FCPlantNoRecirc
  extends Modelica.Icons.Example;

  package Medium_an = FCSys.WorkInProgress.MolarBasis.H2O_H2_CO_Molar 
    "Anode medium model";
  package Medium_ca = FCSys.WorkInProgress.MolarBasis.H2OAndO2_N2_Molar 
    "Cathode medium model";

  replaceable Modelica.Blocks.Sources.Constant valveSetpoint(k=0.5) 
    constrainedby Modelica.Blocks.Interfaces.SO;
  FCSys.WorkInProgress.Systems.FC.FCNoRecirc fCPlant;
  Modelica.Fluid.Sources.FixedBoundary_pTX ambientCa(
    redeclare replaceable Medium_ca Medium,
    T=ambient.default_T_ambient,
    flowDirection=Modelica.Fluid.Types.SourceFlowDirection.InToPort,
    p=ambient.p_default_ambient,
    X=Medium_ca.X_default);
  Modelica.Fluid.Sources.FixedBoundary_pTX ambientAn(
    redeclare replaceable Medium_an Medium,
    T=ambient.default_T_ambient,
    flowDirection=Modelica.Fluid.Types.SourceFlowDirection.InToPort,
    p=ambient.p_default_ambient,
    X=Medium_an.X_default);
  replaceable Modelica.Blocks.Sources.Constant fanSpeed(k=0.5) constrainedby 
    Modelica.Blocks.Interfaces.SO;
  inner Modelica.Fluid.Ambient ambient(p_default_ambient=
        SI.Conversions.from_bar(1.01325), default_T_ambient=
        SI.Conversions.from_degC(20));
  Modelica.Thermal.HeatTransport.Sources.FixedTemperature fixedTemperature(T=
        ambient.default_T_ambient);
  Modelica.Electrical.Analog.Basic.Resistor resistor(R=1);
  Modelica.Electrical.Analog.Basic.Ground ground;
  Modelica.Fluid.Pipes.BaseClasses.PortVolume H2tank(
    redeclare FCSys.WorkInProgress.MolarBasis.H2O_H2_CO_Molar Medium,
    V=1e3,
    p_start=10e5,
    use_T_start=true,
    T_start=ambient.default_T_ambient,
    X_start={0,1,0},
    initType=Modelica.Fluid.Types.Init.NoInit);

  Modelica.Thermal.HeatTransport.Species.ThermalConductor thermalConductor2(G=100);
  Modelica.Fluid.Pipes.BaseClasses.PortVolume O2tank(
    V=1e3,
    p_start=10e5,
    use_T_start=true,
    T_start=ambient.default_T_ambient,
    X_start={0,1,0},
    redeclare FCSys.WorkInProgress.MolarBasis.H2OAndO2_N2_Molar Medium,
    initType=Modelica.Fluid.Types.Init.NoInit);

  Modelica.Thermal.HeatTransport.Species.ThermalConductor thermalConductor1(G=100);

equation 
  connect(valveSetpoint.y, fCPlant.exitValvePosAn);
  connect(fCPlant.exitValvePosCa, valveSetpoint.y);
  connect(fanSpeed.y, fCPlant.coolingFanSpeed);
  connect(fCPlant.fluidPort_ca2, ambientCa.port);
  connect(fCPlant.fluidPort_an2, ambientAn.port);
  connect(fixedTemperature.port, fCPlant.port_a);
  connect(resistor.n, fCPlant.n);
  connect(resistor.p, fCPlant.p);
  connect(ground.p, resistor.n);
  connect(H2tank.port, fCPlant.fluidPort_an1);
  connect(thermalConductor2.port_b, H2tank.heatPort);
  connect(thermalConductor2.port_a, fixedTemperature.port);
  connect(thermalConductor1.port_a, fixedTemperature.port);
  connect(thermalConductor1.port_b, O2tank.heatPort);
  connect(O2tank.port, fCPlant.fluidPort_ca1);
end FCPlantNoRecirc;

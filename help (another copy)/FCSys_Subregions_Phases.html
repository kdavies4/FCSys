protected 
  Reaction combustion(n_spec=3, formulas={"H2","O2","H2O"}) if inclReact and (
    inclH2 and inclH2O and inclO2) "2H2 + O2 <=> 2H2O";

  // **Clean up diagram for this and other phases.

equation 
  // Chemical interactions
  connect(combustion.chemical[1], H2.chemical);
  connect(combustion.chemical[2], O2.chemical);
  connect(combustion.chemical[3], H2O.chemical);

  // H2
  // --

  connect(H2.chemical, chemical.H2);
  connect(H2.common.mechanical, common.mechanical);

  // Transport

  connect(H2.faces[Axis.x, Side.p], xPositive.H2);

  connect(H2.faces[Axis.y, Side.n], yNegative.H2);

  connect(H2.faces[Axis.y, Side.p], yPositive.H2);

  // H2O

  connect(H2O.common.mechanical, common.mechanical);
  connect(H2O.common.thermal, common.thermal);
  connect(H2O.inert, phaseBoundary.inertD);
  // Transport
  connect(H2O.faces[Axis.x, Side.n], xNegative.H2O);

  connect(H2O.faces[Axis.x, Side.p], xPositive.H2O);

  connect(H2O.faces[Axis.y, Side.n], yNegative.H2O);

  connect(H2O.faces[Axis.y, Side.p], yPositive.H2O);

  connect(N2.inert, phaseBoundary.inertD);

  connect(N2.faces[Axis.x, Side.p], xPositive.N2);

    then initVelX, initVelY, and initVelZ are

  extends BaseClasses.NullPhase(final n_spec=countTrue({'inclC+','incle-'}),
      common(mechanical(phi(fixed=if 'inclC+' then fill(false, n_lin) else {

    Species.Species(
    initMethX=if (initVelX or 'inclC+') and reduceVel then InitMethVelocity.None
         else InitMethVelocity.Velocity,
    initMethY=if (initVelY or 'inclC+') and reduceVel then InitMethVelocity.None
         else InitMethVelocity.Velocity,
    initMethZ=if (initVelZ or 'inclC+') and reduceVel then InitMethVelocity.None
         else InitMethVelocity.Velocity,
    initMethTemp=if initTemp and reduceTemp then InitMethScalar.None else 
        InitMethScalar.Temperature,
    phi(each stateSelect=if reduceVel then StateSelect.default else StateSelect.prefer),

    T(stateSelect=if reduceTemp then StateSelect.default else StateSelect.prefer))
    "
'e-' model
";

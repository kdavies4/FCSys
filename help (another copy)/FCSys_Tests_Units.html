Test the unit conversions in the Units package

Modelica definition


function testConversions 
  "
Test the unit conversions in the Units package
"

  import FCSys.Test.assertValue;
  import FCSys.Units.*;
  extends Modelica.Icons.Function;

  output Boolean ok "true, if all tests passed";

algorithm 
  ok := false;

  // "From" functions
  assertValue(
    from_degC(100),
    373.15*U.K,
    name="from_degC");
  assertValue(
    from_kPag(101.325),
    2*U.atm,
    name="from_kPag");

  // Inverses
  assertValue(
    to_degC(from_degC(1)),
    1,
    name="to_degC");
  assertValue(
    to_kPag(from_kPag(1)),
    1,
    name="to_kPag");

  ok := true;
end testConversions;

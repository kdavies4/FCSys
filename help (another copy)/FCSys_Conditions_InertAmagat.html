FCSys.Conditions.InertAmagat.Phase

Condition for a InertAmagat connector (e.g., as in a Phase model), with efforts by default

Information


  
If the source of an internal specification is redeclared to a block besides
  Modelica.Blocks.Sources.Constant,
  then the related condition must be redeclared as well.  For example, use:

  redeclare Conditions.InertAmagat.Amagat.Volume amagat(redeclare Modelica.Blocks.Sources.Ramp source)

  rather than simply:

  amagat(redeclare Modelica.Blocks.Sources.Ramp source)

  

Extends from FCSys.Conditions.BaseClasses.Icons.Single (Icon for a single-connector boundary condition).

Modelica definition


model Phase 
  "
Condition for a InertAmagat connector (e.g., as in a Phase model), with efforts by default
"
  extends FCSys.Conditions.BaseClasses.Icons.Single;

  // Included components of linear momentum
  parameter Boolean inclLinX=true "X";

  parameter Boolean inclLinY=false "Y";

  parameter Boolean inclLinZ=false "Z";

  // Conditions
  replaceable Amagat.Pressure amagat(source(k(start=U.atm))) constrainedby 
    Amagat.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ) "Amagat";
  replaceable Mechanical.Velocity mechanicalX(source(k(start=0))) if inclLinX 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.x) "X-axis mechanical";
  replaceable Mechanical.Velocity mechanicalY(source(k(start=0))) if inclLinY 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.y) "Y-axis mechanical";
  replaceable Mechanical.Velocity mechanicalZ(source(k(start=0))) if inclLinZ 
    constrainedby Mechanical.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ,
    final axis=Axis.z) "Z-axis mechanical";
  replaceable Thermal.Temperature thermal(source(k(start=298.15*U.K))) 
    constrainedby Thermal.BaseClasses.PartialCondition(
    final inclLinX=inclLinX,
    final inclLinY=inclLinY,
    final inclLinZ=inclLinZ) "Thermal";

  Connectors.RealInputBus u "Input bus for values of specified conditions";
  Connectors.RealOutputBus y "Output bus of measurements";

  Connectors.InertAmagat inert(final n_lin=countTrue({inclLinX,inclLinY,
        inclLinZ})) 
    "Single-species connector for linear momentum and heat, with additivity of volume";

equation 
  // Amagat
  connect(amagat.inert, inert);
  connect(u.amagat, amagat.u);
  connect(amagat.y, y.amagat);

  // X-axis mechanical
  connect(mechanicalX.inert, inert);
  connect(u.mechanicalX, mechanicalX.u);
  connect(mechanicalX.y, y.mechanicalX);

  // Y-axis mechanical
  connect(mechanicalY.inert, inert);

  connect(u.mechanicalY, mechanicalY.u);
  connect(mechanicalY.y, y.mechanicalY);

  // Z-axis mechanical
  connect(mechanicalZ.inert, inert);
  connect(u.mechanicalZ, mechanicalZ.u);
  connect(mechanicalZ.y, y.mechanicalZ);

  // Thermal
  connect(thermal.inert, inert);
  connect(u.thermal, thermal.u);
  connect(thermal.y, y.thermal);
end Phase;

model PhaseFlow 
  "
Condition for a InertAmagat connector (e.g., as in a Phase model), with flows by default
"

  extends Phase(
    redeclare Amagat.Volume amagat(source(k(start=-U.cc))),
    redeclare Mechanical.Force mechanicalX(source(k(start=0))),
    redeclare Mechanical.Force mechanicalY(source(k(start=0))),
    redeclare Mechanical.Force mechanicalZ(source(k(start=0))),
    redeclare Thermal.HeatRate thermal(source(k(start=0))));

end PhaseFlow;

model ConditionsAdaptersPhasesIonomer 
  "
Adapter for ionomer between FCSys and Modelica
"
  extends Conditions.Adapters.Phases.BaseClasses.PartialPhase;
  extends Modelica.Icons.UnderConstruction;

  Conditions.Adapters.Species.Solid 'C19HF37O5S-';
  FCSys.WorkInProgress.ConditionsAdaptersSpeciesFluid 'H+'(redeclare package
      Data = FCSys.Characteristics.'H+'.Ionomer, redeclare package Medium = Modelica.Media.IdealGases.SingleGases.H2);
    
  // **Use model for H instead.

  Conditions.Adapters.Species.FluidNonionic H2O(redeclare package Data = Characteristics.H2O.Ionomer,
      redeclare package Medium = Modelica.Media.IdealGases.SingleGases.H2O);
  Conditions.Adapters.Junctions.Junction2 junction2;
  Modelica.Fluid.Interfaces.FluidPort_b fluidPort(redeclare final package
      Medium = Medium) "Modelica fluid port";
  Modelica.Electrical.Analog.Interfaces.NegativePin pin 
    "Modelica electrical pin";

equation 
  connect('C19HF37O5S-'.face.thermal, face.'C19HF37O5S-'.thermal);

  connect('H+'.face.normal, face.'H+'.normal);

  connect('H+'.face.thermal, face.'H+'.thermal);

  connect('H+'.pin, pin);

  connect('H+'.heatPort, heatPort);
  connect('C19HF37O5S-'.heatPort, heatPort);
  connect(junction2.mixturePort, fluidPort);
end ConditionsAdaptersPhasesIonomer;
